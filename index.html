// A simple DAX measure
Total Sales = 
SUMX (
    Sales,
    Sales[Quantity] * Sales[Net Price]
)

/* 
 * Multi-line comment 
 * demonstrating a variable
 */
Sales YTD = 
VAR CurrentDate = MAX('Date'[Date])
RETURN
    CALCULATE(
        [Total Sales],
        DATESYTD('Date'[Date])
    )

EVALUATE
  FILTER(
    'Product',
    'Product'[Category] = "Bikes" && 'Product'[ListPrice] > 1000.50
  )
